        self._value = 0

    def inc(self) -> int:
        with self._lock:
            self._value += 1
            return self._value
    @property
    def value(self) -> int:
        with self._lock:
            return self._value

class ApiClient:
    def __init__(self, config: ApiConfig, public_key: str) -> None:
        self.config = config

        self.session.verify = False

            "content-type": config.content_type,

    def close(self) -> None:

    def set_token(self, token: str | None) -> None:
        if token is None:
            self.session.headers.pop("token", None)
        else:
            self.session.headers.update({"token": token})
    def post_plain(self, path: str, payload: dict[str, Any], params: dict[str, Any] | None = None) -> ApiResult:
        return self._post(path, payload, params=params)
    def post_rsa(self, path: str, payload: dict[str, Any]) -> ApiResult:
        return self._post(path, {"encrypt": self.rsa_encrypt_payload(payload)})
    def rsa_encrypt_payload(self, payload: dict[str, Any]) -> str:
        plain = json.dumps(payload, ensure_ascii=False, separators=(",", ":")).encode("utf-8")
        cipher = PKCS1_v1_5.new(self._public_key)
        chunk_size = self._public_key.size_in_bytes() - 11
        encrypted = bytearray()
        for i in range(0, len(plain), chunk_size):
            encrypted.extend(cipher.encrypt(plain[i : i + chunk_size]))
        return base64.b64encode(bytes(encrypted)).decode("ascii")
    def decrypt_response_text(self, raw_text: str) -> dict[str, Any]:
        text = raw_text.strip()
        if text.startswith('"') and text.endswith('"'):
            text = json.loads(text)
        cipher = AES.new(self._aes_key, AES.MODE_ECB)
        plain = unpad(cipher.decrypt(base64.b64decode(text)), AES.block_size).decode("utf-8")
        return json.loads(plain)
    def _post(self, path: str, payload: dict[str, Any], params: dict[str, Any] | None = None) -> ApiResult:
        response = self.session.post(
            f"{self.config.base_url.rstrip('/')}/{path.lstrip('/')}",
            params=params, json=payload, timeout=self.config.timeout,
        )
        response.raise_for_status()
        result = self.decrypt_response_text(response.text)
        return ApiResult(code=result["code"], msg=result["msg"], data=result["data"])
    @staticmethod

def make_uid() -> str:
    raw = f"{time.time()}{random.random()}{threading.get_ident()}"
    uid = hex(hash(raw))[2:]
    return uid if len(uid) == 16 else f"1{uid[:15]}"

def do_invite(refer_code: str, counter: AtomicCounter, total: int, log_callback) -> tuple[bool, str]:

    n = counter.inc()
    log_callback(n, total, success, msg)
    return success, msg

## 它长什么样

套路很经典，一眼能认：

1. **一段 PowerShell 里嵌着一整个 Python 程序**。用 here-string 把一坨代码塞进去，再 `Set-Content` 写到桌面上的某个 `.py`。这是典型的"复制粘贴到终端就能跑"投递手法——你不细看的话，以为只是在装个工具。
2. **开头先装依赖**。脚本一上来就检查 `requests`、`pycryptodome` 缺不缺，缺就自动 `pip install`。在 Windows 上还会先卸载掉互相冲突的 `crypto`、`pycryptodome` 再重装。换句话说，作者已经替你想好了"跑不起来"的所有借口。
3. **关掉一切校验**。`urllib3.disable_warnings()`、`verify=False`，再加一个伪造的 Chrome User-Agent。一个正经工具不需要这些；需要这些的，是怕被对面认出来。
4. **裹一层加密**。请求用 RSA 分块加密，响应用 AES-ECB 解。不是在保护你的数据，是在模仿某个 App 的私有协议，好让服务端把它当成真客户端。

这些特征凑在一起，性质就很清楚了：这是一把对着某个在线服务的"钥匙"，配了一台"批量伪造身份"的机器。

## 它在干什么

剥掉皮，逻辑一共三步，每一步都踩在红线上：

- **批量造假账号**。脚本里有个函数专门生成一次性的随机 ID，然后挨个去服务端"登录"——注册出一堆根本不存在的设备。
- **自己给自己刷推荐**。每个假账号登录之后，立刻把**你填进去的邀请码**绑到自己身上。于是你名下凭空多了一堆"被你邀请来的用户"。
- **并发拉满**。线程池默认开十几个、能到几十，两个按钮直接写明"刷 100 人""刷 200 人"。这不是在测试，是在薅。

翻译成人话：拿假身份把邀请奖励刷到自己兜里。对面服务若是按邀请给会员、给金币、给返利，这套东西干的就是**欺诈**。哪怕那个服务本身就是个灰色地带的软件，薅它也还是薅，不会因为你薅的是"坏东西"就变正当。

## 为什么不能碰

三个理由，按严重程度排：

1. **法律层面**。用伪造身份骗取奖励，往轻了说是违反服务条款、账号封禁，往重了说就是诈骗。IP、时间戳、收益记录全留痕，真要追，跑不掉。"我只是跑了个脚本"不是抗辩理由。
2. **你自己的机器**。一段需要你关掉 TLS 校验、自动卸装重装 crypto 库、往桌面写文件的代码，你凭什么相信它只做它说的那件事？今天它刷邀请，明天它顺手读你的浏览器 cookie、挖你的钱包助记词，你根本看不出来。这种"先装依赖再跑"的投递手法，和恶意软件的加载链长得一模一样。
3. **它可能就是个套**。这类脚本流传的圈子里，夹带后门是常态。你以为是白嫖工具，作者可能把你机器当肉鸡、把你的算力当矿机。免费的最贵。

## 真碰到了怎么清理

万一已经手贱跑过，按这个顺序收拾：

1. **删掉被投递的文件**。桌面、下载目录、临时目录里凡是那段时间冒出来的 `.py`，尤其文件名像 `invite_tool.py` 之类的，全删。别光删桌面那个，搜一遍。
2. **清掉它装的包**。它强制重装了 `pycryptodome`，如果这是它唯一干的事还好；保险起见检查最近装的 pip 包：`pip list` 看看有没有你不认识的。不确定的，虚拟环境里重装一遍你的 Python 环境最干净。
3. **轮换可能泄露的东西**。如果你在跑这段脚本的机器上登过任何重要账号，尤其是同浏览器里登过的，把 cookie/会话当作已泄露处理——重新登录、踢掉其他设备、必要时改密码。
4. **扫一次**。用靠谱的安全软件全盘扫一遍。它关了 TLS 校验、能自动 `pip install`，理论上可以加载任意远端代码，别拿"它看起来没干坏事"赌。

## 怎么一眼识破

记住几个特征，以后再见到类似的，直接划走：

- PowerShell 里用 here-string 嵌一整段代码、再 `Set-Content` 写盘的，**大概率不是好东西**。正经工具不这么投递。
- 脚本一开头就自动 `pip install` 一堆东西、还卸装重装 crypto 库的，警惕。
- 关 TLS 校验、关 urllib3 警告、伪造 UA 的，基本就是在躲对面的风控。
- 界面再卖萌、"温馨提示"写得再诚恳，底层逻辑是"生成假身份 + 绑你的邀请码"，那就是刷量，没得洗。

别人塞给你一段代码让你跑，先问三个问题：它是给谁用的？它要我关掉什么校验？它跑起来之后，谁是受益方？三个答案里只要有任何一个让你迟疑，就别碰。

省下来的那点会员和金币，抵不上一次被封号、被追责、或者机器被种马的代价。
